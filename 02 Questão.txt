SELECT A.NOME, MEDIA , RESULT
CASE
	WHEN MEDIA >= 5 THEN 'APROVADO'
	ELSE'REPROVADO'
	END AS RESULT,

FROM MATRICULADO AS M

INNER JOIN ALUNO AS A
ON M.ALUNO = A.CODIGO

INNER JOIN PROFESSOR AS P
ON M.TURMA = P.CODIGO


WHERE P.NOME =  “Francisco Ferreira Chagas”
WHERE AVG(M.NOTA1  + M.NOTA2 + M.NOTA3) AS MEDIA
